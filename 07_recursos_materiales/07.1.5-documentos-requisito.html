07.1.5

Catálogo de Documentos (Requisitos)

Diccionario maestro de los posibles anexos, oficios y formatos exigibles durante un proceso de adquisición.

Modo A — Catálogo Simple
1 Descripción General

Este catálogo sirve como la base de datos centralizada de todos los tipos de documentos (Ej. "Anexo Técnico", "Cuadro Comparativo", "Estudio de Mercado", "Dictamen de Excepción") que pueden formar parte del Checklist documental de un requerimiento. Posteriormente, en la Matriz de Checklists, estos documentos se enlazarán a las distintas modalidades de compra.

2 Marco Normativo
  • Control Interno: Establece la necesidad de integrar y resguardar expedientes documentales completos como soporte de cualquier erogación de recursos públicos (Auditoría).
3 Roles que intervienen

Definición SQL

-- Función para lectura
CREATE OR REPLACE FUNCTION materiales.fn_leer_documentos_req(
    p_id_documento integer DEFAULT NULL,
    p_busqueda varchar DEFAULT NULL,
    p_estatus integer DEFAULT 1
) RETURNS jsonb LANGUAGE plpgsql AS $$
DECLARE v_resultado jsonb;
BEGIN
    IF p_id_documento > 0 THEN
        SELECT row_to_json(t) INTO v_resultado 
        FROM materiales.cat_documentos_requisito t 
        WHERE t.id_documento = p_id_documento;
    ELSE
        SELECT jsonb_build_object('registros', COALESCE(jsonb_agg(row_to_json(l)), '[]'::jsonb)) INTO v_resultado
        FROM materiales.cat_documentos_requisito l
        WHERE (p_busqueda IS NULL OR l.nombre_documento ILIKE '%' || p_busqueda || '%')
          AND (p_estatus IS NULL OR l.estatus_registro = p_estatus);
    END IF;
    RETURN jsonb_build_object('cod_resultado', 0, 'msg_resultado', 'OK', 'data', v_resultado);
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object('cod_resultado', -1, 'msg_resultado', SQLERRM);
END;
$$;

@p_id_documentoINTRequerido para accion 2 y 3.

Definición SQL

-- Función para ABC
CREATE OR REPLACE FUNCTION materiales.fn_abc_documentos_req(
    p_accion integer,
    p_id_documento integer DEFAULT NULL,
    p_nombre_documento varchar DEFAULT NULL,
    p_es_obligatorio boolean DEFAULT true
) RETURNS jsonb LANGUAGE plpgsql AS $$
DECLARE v_id integer;
BEGIN
    IF p_accion = 1 THEN
        INSERT INTO materiales.cat_documentos_requisito (nombre_documento, es_obligatorio, estatus_registro)
        VALUES (p_nombre_documento, p_es_obligatorio, 1) RETURNING id_documento INTO v_id;
    ELSIF p_accion = 2 THEN
        UPDATE materiales.cat_documentos_requisito 
        SET nombre_documento = p_nombre_documento, es_obligatorio = p_es_obligatorio
        WHERE id_documento = p_id_documento;
        v_id := p_id_documento;
    ELSIF p_accion = 3 THEN
        UPDATE materiales.cat_documentos_requisito SET estatus_registro = 0
        WHERE id_documento = p_id_documento;
        v_id := p_id_documento;
    END IF;
    RETURN jsonb_build_object('cod_resultado', 0, 'msg_resultado', 'Guardado correctamente', 'data', jsonb_build_object('id_documento', v_id));
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object('cod_resultado', -1, 'msg_resultado', SQLERRM);
END;
$$;
