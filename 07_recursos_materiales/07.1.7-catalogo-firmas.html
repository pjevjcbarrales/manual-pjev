07.1.7

Catálogo de Firmas

Administra al empleado responsable de cada rol del módulo de Recursos Materiales, su adscripción, puesto, nombre de presentación y vigencia.

Formulario simpleEsquema: materialesPatrón: LEER + ABC
1 Descripción General

El módulo mantiene una lista única de las personas responsables que pueden intervenir como Revisor, Autorizador, Remitente o Destinatario dentro de Recursos Materiales. Cada registro enlaza un rol predefinido con una organización vigente, un área perteneciente a esa organización, un puesto y un empleado activo de esa misma adscripción.

El campo Nombre con el que firma conserva la presentación que debe utilizarse en documentos, por ejemplo “Dra. Paulina Ochoa Lara”, sin modificar config.catalogo_empleados.ce_nombre. La vigencia inicial la captura el usuario; la vigencia final de un alta queda en 2099-12-31. Para sustituir a una persona se cierra primero la vigencia anterior y después se registra la nueva, sin traslapar fechas.

Alcance: este catálogo no crea documentos, no almacena imágenes de firma y no ejecuta procesos de suficiencia. Otros módulos de Recursos Materiales únicamente consultan quién ocupa cada rol en una fecha determinada.

CREATE EXTENSION IF NOT EXISTS btree_gist;

CREATE TABLE materiales.cat_firmas (
  cf_id BIGSERIAL PRIMARY KEY,
  cf_rol VARCHAR(20) NOT NULL,
  cc_id INTEGER NOT NULL REFERENCES config.catalogo_centros_costos(cc_id),
  cua_id INTEGER NOT NULL REFERENCES config.catalogo_ua(cua_id),
  cf_puesto VARCHAR(100) NOT NULL,
  ce_id INTEGER NOT NULL REFERENCES config.catalogo_empleados(ce_id),
  cf_nombre_firma VARCHAR(200) NOT NULL,
  cf_vigencia_inicio DATE NOT NULL,
  cf_vigencia_fin DATE NOT NULL DEFAULT DATE '2099-12-31',
  estatus_registro SMALLINT NOT NULL DEFAULT 1,
  usuario_c VARCHAR(20) NOT NULL,
  fecha_c TIMESTAMPTZ(6) NOT NULL DEFAULT CURRENT_TIMESTAMP,
  usuario_m VARCHAR(20), fecha_m TIMESTAMPTZ(6),
  CONSTRAINT ck_firma_rol CHECK
    (cf_rol IN ('REVISOR','AUTORIZADOR','REMITENTE','DESTINATARIO')),
  CONSTRAINT ck_firma_fechas CHECK (cf_vigencia_inicio <= cf_vigencia_fin),
  CONSTRAINT ck_firma_estatus CHECK (estatus_registro IN (0,1)),
  CONSTRAINT ck_firma_textos CHECK
    (BTRIM(cf_puesto) <> '' AND BTRIM(cf_nombre_firma) <> '')
);

ALTER TABLE materiales.cat_firmas ADD CONSTRAINT ex_firma_rol_sin_traslape
EXCLUDE USING gist (

CREATE INDEX ix_firmas_empleado ON materiales.cat_firmas(ce_id);
CREATE INDEX ix_firmas_adscripcion ON materiales.cat_firmas(cc_id,cua_id);

CREATE OR REPLACE FUNCTION materiales.fn_leer_catalogo_firmas(
 p_cf_id BIGINT DEFAULT NULL, p_busqueda VARCHAR DEFAULT NULL,
 p_rol VARCHAR DEFAULT NULL, p_fecha DATE DEFAULT CURRENT_DATE
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_data JSONB;
BEGIN
 IF p_cf_id = -1 THEN
  v_data := jsonb_build_object(
   'registro',jsonb_build_object('cf_id',0,'cf_rol',NULL,'cc_id',NULL,
    'cua_id',NULL,'cf_puesto','','ce_id',NULL,'cf_nombre_firma','',
    'cf_vigencia_inicio',CURRENT_DATE,'cf_vigencia_fin',DATE '2099-12-31'),
   'roles',jsonb_build_array('REVISOR','AUTORIZADOR','REMITENTE','DESTINATARIO'));
 ELSIF p_cf_id > 0 THEN
  SELECT to_jsonb(t) INTO v_data FROM (
   SELECT f.*,cc.cc_clave_pptal,cc.cc_nombre AS organizacion,
    u.cua_numua,u.cua_nombre AS area,e.ce_numemp,e.ce_nombre AS empleado
   FROM materiales.cat_firmas f
   JOIN config.catalogo_centros_costos cc ON cc.cc_id=f.cc_id
   JOIN config.catalogo_ua u ON u.cua_id=f.cua_id
   JOIN config.catalogo_empleados e ON e.ce_id=f.ce_id
   WHERE f.cf_id=p_cf_id) t;
  IF v_data IS NULL THEN
   RETURN jsonb_build_object('cod_resultado',-1,'msg_resultado','La firma no existe.','data',NULL);
  END IF;
 ELSE
  SELECT jsonb_build_object('registros',COALESCE(jsonb_agg(to_jsonb(t)),'[]'::jsonb))
  INTO v_data FROM (
   SELECT f.cf_id,f.cf_rol,f.cf_nombre_firma,e.ce_numemp,e.ce_nombre AS empleado,
    cc.cc_nombre AS organizacion,u.cua_nombre AS area,f.cf_puesto,
    f.cf_vigencia_inicio,f.cf_vigencia_fin,
    CASE WHEN f.estatus_registro=0 THEN 'BAJA'
     WHEN p_fecha NOT BETWEEN f.cf_vigencia_inicio AND f.cf_vigencia_fin THEN 'FUERA_DE_VIGENCIA'
     WHEN e.estatus_registro<>1 OR cc.estatus_registro<>1 OR u.estatus_registro<>1
      OR e.ce_cc_id IS DISTINCT FROM f.cc_id OR e.ce_ua IS DISTINCT FROM f.cua_id
      OR UPPER(BTRIM(e.ce_puesto))<>UPPER(BTRIM(f.cf_puesto))
      OR p_fecha NOT BETWEEN COALESCE(cc.cc_vigencia_inicio,DATE '1900-01-01')
                         AND COALESCE(cc.cc_vigencia_fin,DATE '2099-12-31')
     THEN 'NO_ELEGIBLE' ELSE 'VIGENTE' END AS vigencia
   FROM materiales.cat_firmas f
   JOIN config.catalogo_centros_costos cc ON cc.cc_id=f.cc_id
   JOIN config.catalogo_ua u ON u.cua_id=f.cua_id
   JOIN config.catalogo_empleados e ON e.ce_id=f.ce_id
   WHERE (p_rol IS NULL OR f.cf_rol=UPPER(BTRIM(p_rol)))
    AND (p_busqueda IS NULL OR concat_ws(' ',f.cf_nombre_firma,e.ce_numemp,
     e.ce_nombre,f.cf_puesto,cc.cc_nombre,u.cua_nombre) ILIKE '%'||p_busqueda||'%')
   ORDER BY f.cf_rol,f.cf_vigencia_inicio DESC) t;
 END IF;
 RETURN jsonb_build_object('cod_resultado',0,'msg_resultado','OK','data',v_data);
EXCEPTION WHEN OTHERS THEN
 RETURN jsonb_build_object('cod_resultado',-2,'msg_resultado',SQLERRM,'data',NULL);
END; $$;

Acciones: 1=Alta, 2=Cambio, 3=Baja lógica. p_usuario se obtiene del JWT.

CREATE OR REPLACE FUNCTION materiales.fn_abc_catalogo_firmas(
 p_accion SMALLINT,p_cf_id BIGINT DEFAULT NULL,p_cf_rol VARCHAR DEFAULT NULL,
 p_cc_id INT DEFAULT NULL,p_cua_id INT DEFAULT NULL,p_cf_puesto VARCHAR DEFAULT NULL,
 p_ce_id INT DEFAULT NULL,p_cf_nombre_firma VARCHAR DEFAULT NULL,
 p_cf_vigencia_inicio DATE DEFAULT NULL,
 p_cf_vigencia_fin DATE DEFAULT DATE '2099-12-31',
 p_usuario VARCHAR DEFAULT NULL
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_id BIGINT; v_actual materiales.cat_firmas%ROWTYPE;
BEGIN
 IF p_accion NOT IN (1,2,3) OR NULLIF(BTRIM(p_usuario),'') IS NULL THEN
  RETURN jsonb_build_object('cod_resultado',-1,'msg_resultado','Acción o usuario inválido.','data',NULL);
 END IF;
 IF p_accion=1 THEN
  IF p_cf_vigencia_fin IS DISTINCT FROM DATE '2099-12-31' THEN
   RETURN jsonb_build_object('cod_resultado',-1,'msg_resultado','Una alta debe terminar el 31/12/2099.','data',NULL);
  END IF;
  INSERT INTO materiales.cat_firmas(cf_rol,cc_id,cua_id,cf_puesto,ce_id,
   cf_nombre_firma,cf_vigencia_inicio,cf_vigencia_fin,usuario_c)
  VALUES(UPPER(BTRIM(p_cf_rol)),p_cc_id,p_cua_id,BTRIM(p_cf_puesto),p_ce_id,
   BTRIM(p_cf_nombre_firma),p_cf_vigencia_inicio,DATE '2099-12-31',LEFT(p_usuario,20))
  RETURNING cf_id INTO v_id;
 ELSIF p_accion=2 THEN
  SELECT * INTO v_actual FROM materiales.cat_firmas WHERE cf_id=p_cf_id FOR UPDATE;
  IF NOT FOUND THEN RETURN jsonb_build_object('cod_resultado',-1,'msg_resultado','La firma no existe.','data',NULL); END IF;
  IF v_actual.cf_vigencia_inicio<=CURRENT_DATE AND
   (UPPER(BTRIM(p_cf_rol)) IS DISTINCT FROM v_actual.cf_rol
    OR p_cc_id IS DISTINCT FROM v_actual.cc_id OR p_cua_id IS DISTINCT FROM v_actual.cua_id
    OR UPPER(BTRIM(p_cf_puesto)) IS DISTINCT FROM UPPER(BTRIM(v_actual.cf_puesto))
    OR p_ce_id IS DISTINCT FROM v_actual.ce_id
    OR p_cf_vigencia_inicio IS DISTINCT FROM v_actual.cf_vigencia_inicio) THEN
   RETURN jsonb_build_object('cod_resultado',-1,
    'msg_resultado','Una vigencia iniciada sólo permite cambiar nombre de firma o fecha final. Cierre y registre la sustitución.','data',NULL);
  END IF;
  UPDATE materiales.cat_firmas SET cf_rol=UPPER(BTRIM(p_cf_rol)),cc_id=p_cc_id,
   cua_id=p_cua_id,cf_puesto=BTRIM(p_cf_puesto),ce_id=p_ce_id,
   cf_nombre_firma=BTRIM(p_cf_nombre_firma),cf_vigencia_inicio=p_cf_vigencia_inicio,
   cf_vigencia_fin=p_cf_vigencia_fin,usuario_m=LEFT(p_usuario,20),fecha_m=CURRENT_TIMESTAMP
  WHERE cf_id=p_cf_id RETURNING cf_id INTO v_id;
 ELSE
  UPDATE materiales.cat_firmas SET estatus_registro=0,usuario_m=LEFT(p_usuario,20),
   fecha_m=CURRENT_TIMESTAMP WHERE cf_id=p_cf_id AND estatus_registro=1 RETURNING cf_id INTO v_id;
  IF v_id IS NULL THEN RETURN jsonb_build_object('cod_resultado',-1,'msg_resultado','La firma no existe o ya está de baja.','data',NULL); END IF;
 END IF;
 RETURN jsonb_build_object('cod_resultado',0,'msg_resultado','Operación realizada correctamente.','data',jsonb_build_object('cf_id',v_id));
EXCEPTION WHEN exclusion_violation THEN
 RETURN jsonb_build_object('cod_resultado',-1,'msg_resultado','Ya existe el mismo rol con vigencia traslapada.','data',NULL);
WHEN SQLSTATE 'P0001' THEN
 RETURN jsonb_build_object('cod_resultado',-1,'msg_resultado',SQLERRM,'data',NULL);
WHEN OTHERS THEN
 RETURN jsonb_build_object('cod_resultado',-2,'msg_resultado',SQLERRM,'data',NULL);
END; $$;

CREATE OR REPLACE FUNCTION materiales.fn_lista_puestos_firma(p_cc_id INT,p_cua_id INT)
RETURNS JSONB LANGUAGE sql STABLE AS $$
 SELECT COALESCE(jsonb_agg(to_jsonb(t) ORDER BY t.puesto),'[]'::jsonb)
 FROM (SELECT DISTINCT BTRIM(ce_puesto) AS puesto FROM config.catalogo_empleados
  WHERE estatus_registro=1 AND ce_cc_id=p_cc_id AND ce_ua=p_cua_id
   AND NULLIF(BTRIM(ce_puesto),'') IS NOT NULL) t; $$;

CREATE OR REPLACE FUNCTION materiales.fn_lista_empleados_firma(
 p_cc_id INT,p_cua_id INT,p_puesto VARCHAR,p_busqueda VARCHAR DEFAULT NULL
) RETURNS JSONB LANGUAGE sql STABLE AS $$
 SELECT COALESCE(jsonb_agg(to_jsonb(t) ORDER BY t.nombre),'[]'::jsonb)
 FROM (SELECT ce_id AS id,ce_numemp AS numero,ce_nombre AS nombre,ce_puesto AS puesto
  FROM config.catalogo_empleados WHERE estatus_registro=1
   AND ce_cc_id=p_cc_id AND ce_ua=p_cua_id
   AND UPPER(BTRIM(ce_puesto))=UPPER(BTRIM(p_puesto))
   AND (p_busqueda IS NULL OR concat_ws(' ',ce_numemp,ce_nombre) ILIKE '%'||p_busqueda||'%')) t; $$;
CREATE OR REPLACE FUNCTION materiales.fn_validar_catalogo_firmas()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE v_cc config.catalogo_centros_costos%ROWTYPE;
 v_emp config.catalogo_empleados%ROWTYPE;
BEGIN
 IF NEW.estatus_registro=0 THEN RETURN NEW; END IF;
 NEW.cf_rol:=UPPER(BTRIM(NEW.cf_rol)); NEW.cf_puesto:=BTRIM(NEW.cf_puesto);
 NEW.cf_nombre_firma:=BTRIM(NEW.cf_nombre_firma);
 SELECT * INTO v_cc FROM config.catalogo_centros_costos WHERE cc_id=NEW.cc_id;
 IF NOT FOUND OR v_cc.estatus_registro<>1 OR
  NEW.cf_vigencia_inicio NOT BETWEEN COALESCE(v_cc.cc_vigencia_inicio,DATE '1900-01-01')
   AND COALESCE(v_cc.cc_vigencia_fin,DATE '2099-12-31') THEN
  RAISE EXCEPTION 'La organización no está vigente en la fecha inicial.' USING ERRCODE='P0001';
 END IF;
 IF NOT EXISTS(SELECT 1 FROM config.catalogo_ua u WHERE u.cua_id=NEW.cua_id
  AND u.estatus_registro=1 AND (u.cua_clavepptal=v_cc.cc_clave_pptal OR v_cc.cc_ua_id=u.cua_id)) THEN
  RAISE EXCEPTION 'El área no pertenece a la organización o no está activa.' USING ERRCODE='P0001';
 END IF;
 SELECT * INTO v_emp FROM config.catalogo_empleados WHERE ce_id=NEW.ce_id;
 IF NOT FOUND OR v_emp.estatus_registro<>1 OR v_emp.ce_cc_id IS DISTINCT FROM NEW.cc_id
  OR v_emp.ce_ua IS DISTINCT FROM NEW.cua_id
  OR UPPER(BTRIM(v_emp.ce_puesto))<>UPPER(NEW.cf_puesto) THEN
  RAISE EXCEPTION 'El empleado no pertenece a la organización, área y puesto.' USING ERRCODE='P0001';
 END IF;
 RETURN NEW;
END; $$;
CREATE TRIGGER trg_validar_catalogo_firmas
BEFORE INSERT OR UPDATE OF cf_rol,cc_id,cua_id,cf_puesto,ce_id,
 cf_nombre_firma,cf_vigencia_inicio,cf_vigencia_fin,estatus_registro
ON materiales.cat_firmas FOR EACH ROW EXECUTE FUNCTION materiales.fn_validar_catalogo_firmas();
