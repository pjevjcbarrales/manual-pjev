Empleados

Plantilla laboral activa e inactiva del PJEV.

Modo A — existe en código Ruta: cfg / catalogos / empleados

Catálogo centralizado del personal de la institución. Enlaza a los empleados con procesos de Nómina, Viáticos, Asignación de Bienes Patrimoniales y Usuarios de Sistema.

Definición SQL (PostgreSQL)

-- ==========================================
-- 1. LEER EMPLEADOS
-- ==========================================
CREATE OR REPLACE FUNCTION config.fn_leer_catalogo_empleados(
    p_organizacion varchar DEFAULT NULL,
    p_unidad_administrativa integer DEFAULT NULL,
    p_filtro varchar DEFAULT NULL
) RETURNS jsonb LANGUAGE plpgsql AS $$
DECLARE v_resultado jsonb;
BEGIN
    SELECT jsonb_build_object('registros', COALESCE(jsonb_agg(row_to_json(t)), '[]'::jsonb)) INTO v_resultado
    FROM (SELECT * FROM config.catalogo_empleados WHERE estatus_registro = 1 
          AND (p_unidad_administrativa IS NULL OR ce_ua = p_unidad_administrativa)
          AND (p_filtro IS NULL OR ce_nombre ILIKE '%' || p_filtro || '%' OR ce_rfc ILIKE '%' || p_filtro || '%' OR ce_curp ILIKE '%' || p_filtro || '%')) t;

    RETURN jsonb_build_object('cod_resultado', 0, 'msg_resultado', 'OK', 'data', v_resultado);
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object('cod_resultado', -2, 'msg_resultado', SQLERRM, 'data', NULL);
END;
$$;

Definición SQL (PostgreSQL)

-- ==========================================
-- 2. ABC EMPLEADOS
-- ==========================================
CREATE OR REPLACE FUNCTION config.fn_abc_catalogo_empleados(
    p_accion integer, p_ejercicio integer, p_usuario varchar, p_ce_id integer DEFAULT NULL,
    p_ce_numemp varchar DEFAULT NULL, p_ce_nombre varchar DEFAULT NULL, p_ce_rfc varchar DEFAULT NULL, 
    p_ce_curp varchar DEFAULT NULL, p_ce_ua integer DEFAULT NULL, p_ce_puesto varchar DEFAULT NULL, 
    p_ce_tipo_contrato smallint DEFAULT NULL, p_ce_clabe varchar DEFAULT NULL, p_ce_banco_id integer DEFAULT NULL, 
    p_ce_email_inst varchar DEFAULT NULL, p_ce_categoria varchar DEFAULT NULL
) RETURNS jsonb LANGUAGE plpgsql AS $$
DECLARE v_id integer := p_ce_id;
BEGIN
    IF p_accion = 1 THEN
        INSERT INTO config.catalogo_empleados(ce_numemp, ce_nombre, ce_rfc, ce_curp, ce_ua, ce_puesto, ce_tipo_contrato, ce_clabe, ce_banco_id, ce_email_inst, ce_categoria, estatus_registro, usuario_c)
        VALUES (p_ce_numemp, p_ce_nombre, p_ce_rfc, p_ce_curp, p_ce_ua, p_ce_puesto, p_ce_tipo_contrato, p_ce_clabe, p_ce_banco_id, p_ce_email_inst, p_ce_categoria, 1, p_usuario) RETURNING ce_id INTO v_id;
    ELSIF p_accion = 2 THEN
        UPDATE config.catalogo_empleados SET ce_numemp = p_ce_numemp, ce_nombre = p_ce_nombre, ce_rfc = p_ce_rfc, ce_curp = p_ce_curp, ce_ua = p_ce_ua, ce_puesto = p_ce_puesto, ce_tipo_contrato = p_ce_tipo_contrato, ce_clabe = p_ce_clabe, ce_banco_id = p_ce_banco_id, ce_email_inst = p_ce_email_inst, ce_categoria = p_ce_categoria, usuario_m = p_usuario, fecha_m = now() WHERE ce_id = p_ce_id;
    ELSIF p_accion = 3 THEN
        UPDATE config.catalogo_empleados SET estatus_registro = 0, usuario_m = p_usuario, fecha_m = now() WHERE ce_id = p_ce_id;
    END IF;
    RETURN jsonb_build_object('cod_resultado', 0, 'msg_resultado', 'Guardado', 'data', jsonb_build_object('id', v_id));
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object('cod_resultado', -2, 'msg_resultado', SQLERRM);
END;
$$;
